10  /**
11   * This is the interface between the agent and the Crypto/Certificate services
12   * available on the software product.
13   * <p>
14   * Brief description of how the Certificates are used: The CSR request is sent
15   * to backend after generating a key pair. The response received will have the
16   * Node Certificate, Signing Certificate and the Licensing SubCa. The Signing
17   * Certificate also known as MMI Respone Signing Certificate. The Node
18   * Certificate a.k.a ID Certificate is the MMI Request Signing Certificate.
19   * <p>
20   * MMI Request messages from the Smart Agent are signed using the Node
21   * Certificate's private key and backend will verify the signature on them to
22   * prove they were not alter enroute.
23   * <p>
24   * MMI response message from backend are signed using the Signing Certificate's
25   * private key and the Smart Agent verified the signature on the message using the
26   * Signing Certificate previously received as a part of CSR process.
27   * 
28   */

36          MD5,
37          /** SHA1 is to be used for the hash. */
38          SHA1,
39          /** SHA256 is to be used for the hash. */
40          SHA256;
41      }

52      public void init(String rootCA);

123     /**
124      * Validates Node Certificate with the root CA.
125      * <p>
126      * Use the method to validate Node Certificate with the root CA. The Node
127      * Certificate is signed with Licensing Sub CA certificate which in turns is
128      * signed using the Root CA certificate. So the certificate chain has to be
129      * validated. Also a double check is done to make sure the public key
130      * contained in the Node certificate is the other half of the key pair
131      * generate as part of CSR by signing a test message with private key and
132      * verifying the signature using the public key present in the Node
133      * certificate.
134      * 
135      * @param cert
136      *            The Node certificate to use for validation in any of the
137      *            formats of DER, Base64, PEM.
138      * @param subCA
139      *            The Licensing SubCA Cert to use in any of the formats of DER,
140      *            Base64 and PEM.
141      * @param privateKey
142      *            The private key stored in trust storage by caller is passed
143      *            else will be null.
144      * @return True if the cert is valid; false otherwise.
145      */
146     public boolean validateNodeCertificate(String cert, String subCA,
147             String privateKey);

149     /**

152      * <p>

156      * 
157      * @param cert

162      * @param signature

164      * @param algorithm
165      *            The message digest algorithm to be used while signing the CSR.

168     public boolean validateSignature(String cert, String data,
